        function formatRupiah(angka) {
            return angka.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ".");
        }

        function renderTable() {
            const tbody = document.getElementById('table-body');
            const tfoot = document.getElementById('table-footer');
            tbody.innerHTML = '';
            tfoot.innerHTML = '';

            let totalPajak = 0;
            let totalPenerimaan = 0;
            let totalSisa = 0;

            data.forEach(row => {
                totalPajak += row.pajak;
                totalPenerimaan += row.penerimaan;
                totalSisa += row.sisa;

                let persenColor = '';
                if (row.persen > 80) {
                    persenColor = 'color: #22c55e; font-weight: bold;';
                } else {
                    persenColor = 'color: #ef4444; font-weight: bold;';
                }

                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td class="border-2 px-3 py-2 text-center">${row.no}</td>
                    <td class="border-2 px-3 py-2">${row.desa}</td>
                    <td class="border-2 px-3 py-2 text-right">${formatRupiah(row.pajak)}</td>
                    <td class="border-2 px-3 py-2 text-right">${formatRupiah(row.penerimaan)}</td>
                    <td class="border-2 px-3 py-2 text-center" style="${persenColor}">${row.persen}</td>
                    <td class="border-2 px-3 py-2 text-right">${formatRupiah(row.sisa)}</td>
                    <td class="border-2 px-3 py-2 text-center" style="${persenColor}">${row.persen}</td>
                    <td class="border-2 px-3 py-2"></td>
                `;
                tbody.appendChild(tr);
            });

            const totalPersen = Math.round((totalPenerimaan / totalPajak) * 100);

            const totalRow = document.createElement('tr');
            totalRow.innerHTML = `
                <td colspan="2" class="border-2 px-3 py-2 text-center font-bold">JUMLAH</td>
                <td class="border-2 px-3 py-2 text-right font-bold">${formatRupiah(totalPajak)}</td>
                <td class="border-2 px-3 py-2 text-right font-bold">${formatRupiah(totalPenerimaan)}</td>
                <td class="border-2 px-3 py-2 text-center font-bold">${totalPersen}</td>
                <td class="border-2 px-3 py-2 text-right font-bold">${formatRupiah(totalSisa)}</td>
                <td class="border-2 px-3 py-2 text-center font-bold">${totalPersen}</td>
                <td class="border-2 px-3 py-2"></td>
            `;
            tfoot.appendChild(totalRow);
        }
