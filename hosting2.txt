    <script>
        let currentStep = 1;
        const totalSteps = 6;
        
        const stepTitles = {
            1: "Persiapan Awal",
            2: "Pilih Penyedia Hosting", 
            3: "Beli Hosting & Domain",
            4: "Install WordPress",
            5: "Setup Awal WordPress",
            6: "Buat Konten Pertama"
        };
        
        function updateProgress() {
            const progress = ((currentStep - 1) / (totalSteps - 1)) * 100;
            document.getElementById('progress-fill').style.width = progress + '%';
            document.getElementById('current-step-text').textContent = stepTitles[currentStep];
        }
        
        function showStep(stepNumber) {
            // Hide all steps
            for (let i = 1; i <= totalSteps; i++) {
                const step = document.getElementById(`step-${i}`);
                step.classList.remove('active', 'completed');
                
                if (i < stepNumber) {
                    step.classList.add('completed');
                } else if (i === stepNumber) {
                    step.classList.add('active');
                }
            }
            
            updateProgress();
        }
        
        function nextStep(stepNumber) {
            if (stepNumber) {
                currentStep = stepNumber + 1;
            } else {
                currentStep++;
            }
            
            if (currentStep <= totalSteps) {
                showStep(currentStep);
            }
        }
        
        function prevStep(stepNumber) {
            if (stepNumber) {
                currentStep = stepNumber - 1;
            } else {
                currentStep--;
            }
            
            if (currentStep >= 1) {
                showStep(currentStep);
            }
        }
        
        function completeGuide() {
            alert('🎉 Selamat! Website WordPress Anda sudah siap digunakan!\n\nSelanjutnya:\n• Buat konten berkualitas secara rutin\n• Promosikan website di media sosial\n• Monitor performa dengan Google Analytics\n• Backup website secara berkala');
            
            // Mark all steps as completed
            for (let i = 1; i <= totalSteps; i++) {
                document.getElementById(`step-${i}`).classList.add('completed');
            }
            
            document.getElementById('progress-fill').style.width = '100%';
            document.getElementById('current-step-text').textContent = 'Website Selesai! 🎉';
        }
        
        // Handle checkbox interactions
        document.addEventListener('change', function(e) {
            if (e.target.type === 'checkbox') {
                const item = e.target.closest('.checklist-item');
                if (e.target.checked) {
                    item.classList.add('completed');
                } else {
                    item.classList.remove('completed');
                }
            }
        });
        
        // Initialize
        showStep(currentStep);
    </script>
