    // Fetch data from Google Sheets CSV Export
    async function fetchGoogleSheetsData() {
      const csvUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vRRQy6LGY1MMNQNINfsA9--x3StH0Rp5oTzM520pjbZuCstJzADBH3RxidcEaKXuYBgwJ9s0fbY_XZG/pub?gid=125225049&single=true&output=csv';
      
      try {
        document.getElementById('loading-overlay').classList.remove('hidden');
        
        const response = await fetch(csvUrl);
        const csv = await response.text();
        
        // Parse CSV
        const lines = csv.split('\n');
        if (lines.length < 2) {
          throw new Error('Data tidak ditemukan');
        }

        const headers = lines[0].split(',').map(h => h.trim().replace(/"/g, ''));

        const data = [];
        for (let i = 1; i < lines.length; i++) {
          const line = lines[i].trim();
          if (!line) continue;

          const cells = [];
          let current = '';
          let inQuotes = false;
          
          for (let j = 0; j < line.length; j++) {
            const char = line[j];
            const nextChar = line[j + 1];
            
            if (char === '"') {
              if (inQuotes && nextChar === '"') {
                current += '"';
                j++;
              } else {
                inQuotes = !inQuotes;
              }
            } else if (char === ',' && !inQuotes) {
              cells.push(current.trim());
              current = '';
            } else {
              current += char;
            }
          }
          cells.push(current.trim());

          const rowData = {};
          headers.forEach((header, index) => {
            rowData[header] = (cells[index] || '').replace(/"/g, '').trim();
          });
          
          if (rowData['ID Keluarga P3KE']) {
            data.push(rowData);
          }
        }

        if (data.length === 0) {
          throw new Error('Tidak ada data yang valid ditemukan');
        }

        allData = data;
        filteredData = [...allData];
        
        init();
        document.getElementById('data-status').innerHTML = `✅ ${data.length} Data Siap`;
        document.getElementById('data-status').classList.remove('bg-sky-100', 'text-sky-800');
        document.getElementById('data-status').classList.add('bg-emerald-100', 'text-emerald-800');
        
      } catch (error) {
        console.error('Error fetching data:', error);
        document.getElementById('data-status').innerHTML = '❌ Gagal Memuat Data';
        document.getElementById('data-status').classList.remove('bg-sky-100', 'text-sky-800');
        document.getElementById('data-status').classList.add('bg-red-100', 'text-red-800');
        showToast('Gagal memuat data: ' + error.message, 'error');
      } finally {
        document.getElementById('loading-overlay').classList.add('hidden');
      }
    }

    // Apply filters
    function applyFilters() {
      const searchId = document.getElementById('search-id').value.toLowerCase();
      const searchName = document.getElementById('search-name').value.toLowerCase();
      const filterKecamatan = document.getElementById('filter-kecamatan').value;
      const filterDesa = document.getElementById('filter-desa').value;
      const filterDesil = document.getElementById('filter-desil').value;
      const filterStatus = document.getElementById('filter-status').value;
      const filterJK = document.getElementById('filter-jk').value;

      filteredData = allData.filter(item => {
        const matchId = !searchId || (item['ID Keluarga P3KE'] || '').toLowerCase().includes(searchId);
        const matchName = !searchName || (item['Nama'] || '').toLowerCase().includes(searchName);
        const matchKecamatan = !filterKecamatan || item['Kecamatan'] === filterKecamatan;
        const matchDesa = !filterDesa || item['Desa/Kelurahan'] === filterDesa;
        const matchDesil = !filterDesil || item['Desil Kesejahteraan'] === filterDesil;
        const matchStatus = !filterStatus || (item[`Penerima ${filterStatus}`] || '').toLowerCase() === 'ya';
        const matchJK = !filterJK || item['Jenis Kelamin'] === filterJK;

        return matchId && matchName && matchKecamatan && matchDesa && matchDesil && matchStatus && matchJK;
      });

      if (sortColumn) {
        filteredData.sort((a, b) => {
          let valA = a[sortColumn] || '';
          let valB = b[sortColumn] || '';
          if (typeof valA === 'string') valA = valA.toLowerCase();
          if (typeof valB === 'string') valB = valB.toLowerCase();
          if (valA < valB) return sortDirection === 'asc' ? -1 : 1;
          if (valA > valB) return sortDirection === 'asc' ? 1 : -1;
          return 0;
        });
      }

      currentPage = 1;
      updateStats();
      renderTable();
    }

    // Get status bantuan
    function getStatusBantuan(item) {
      const statuses = [];
      const statusKeys = ['Penerima BPNT', 'Penerima BPUM', 'Penerima BST', 'Penerima PKH', 'Penerima SEMBAKO'];
      statusKeys.forEach(key => {
        if ((item[key] || '').toLowerCase() === 'ya') {
          statuses.push(key.replace('Penerima ', ''));
        }
      });
      return statuses.length > 0 ? statuses : ['Tidak Ada'];
    }

      const ws = XLSX.utils.aoa_to_sheet([headers, ...rows]);
      const wb = XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb, ws, 'Data P3KE');

      const headerStyle = {
        fill: { fgColor: { rgb: 'FF0C4A6E' } },
        font: { color: { rgb: 'FFFFFFFF' }, bold: true },
        alignment: { horizontal: 'center', vertical: 'center' }
      };

      for (let i = 0; i < headers.length; i++) {
        const cellAddress = XLSX.utils.encode_cell({ r: 0, c: i });
        ws[cellAddress].fill = headerStyle.fill;
        ws[cellAddress].font = headerStyle.font;
        ws[cellAddress].alignment = headerStyle.alignment;
      }

      const colWidths = [18, 20, 15, 15, 15, 18, 10, 25, 15, 15, 15, 12, 8, 8, 8, 8, 10];
      ws['!cols'] = colWidths.map(width => ({ wch: width }));

      const filename = `Data_P3KE_${new Date().toISOString().split('T')[0]}.xlsx`;
      XLSX.writeFile(wb, filename);
      
      showToast(`${filteredData.length} data berhasil diekspor ke Excel`, 'success');
    }

    function bindEvents() {
      document.getElementById('btn-search').onclick = applyFilters;
      document.getElementById('btn-reset').onclick = () => {
        document.getElementById('search-id').value = '';
        document.getElementById('search-name').value = '';
        document.getElementById('filter-kecamatan').value = '';
        document.getElementById('filter-desa').value = '';
        document.getElementById('filter-desil').value = '';
        document.getElementById('filter-status').value = '';
        document.getElementById('filter-jk').value = '';
        sortColumn = null;
        sortDirection = 'asc';
        filteredData = [...allData];
        currentPage = 1;
        updateStats();
        renderTable();
      };
      document.getElementById('btn-export-csv').onclick = exportCSV;
      document.getElementById('btn-export-excel').onclick = exportExcel;

      document.getElementById('per-page').onchange = (e) => {
        perPage = parseInt(e.target.value);
        currentPage = 1;
        renderTable();
      };

      document.getElementById('btn-first').onclick = () => { currentPage = 1; renderTable(); };
      document.getElementById('btn-prev').onclick = () => { if (currentPage > 1) { currentPage--; renderTable(); } };
      document.getElementById('btn-next').onclick = () => { 
        const totalPages = Math.ceil(filteredData.length / perPage);
        if (currentPage < totalPages) { currentPage++; renderTable(); } 
      };
      document.getElementById('btn-last').onclick = () => { 
        currentPage = Math.ceil(filteredData.length / perPage) || 1; 
        renderTable(); 
      };

      document.getElementById('close-modal').onclick = () => {
        document.getElementById('detail-modal').classList.add('hidden');
      };
      document.getElementById('detail-modal').onclick = (e) => {
        if (e.target.id === 'detail-modal') {
          document.getElementById('detail-modal').classList.add('hidden');
        }
      };

      document.querySelectorAll('[data-sort]').forEach(th => {
        th.onclick = () => {
          const col = th.dataset.sort;
          if (sortColumn === col) {
            sortDirection = sortDirection === 'asc' ? 'desc' : 'asc';
          } else {
            sortColumn = col;
            sortDirection = 'asc';
          }
          applyFilters();
        };
      });

      document.getElementById('search-id').onkeyup = (e) => { if (e.key === 'Enter') applyFilters(); };
      document.getElementById('search-name').onkeyup = (e) => { if (e.key === 'Enter') applyFilters(); };

      document.getElementById('filter-kecamatan').onchange = (e) => {
        const selectedKec = e.target.value;
        const desaSelect = document.getElementById('filter-desa');
        const currentDesa = desaSelect.value;
        
        desaSelect.innerHTML = '<option value="">Semua Desa</option>';
        
        const desaList = selectedKec 
          ? [...new Set(allData.filter(d => d['Kecamatan'] === selectedKec).map(d => d['Desa/Kelurahan']).filter(Boolean))]
          : [...new Set(allData.map(d => d['Desa/Kelurahan']).filter(Boolean))];
        
        desaList.sort().forEach(desa => {
          const option = document.createElement('option');
          option.value = desa;
          option.textContent = desa;
          if (desa === currentDesa) option.selected = true;
          desaSelect.appendChild(option);
        });
      };
    }
